//水仙数
#include<stdio.h>
int main() {
    int i,j,k,n;
    printf("水仙数:");
    for(n=100; n<1000; n++) {
        i=n/100;
        j=n/10%10;
        k=n%10;
        if(i*100+j*10+k==i*i*i+j*j*j+k*k*k) {
            printf("%d\n",n);
        }
    }
    return 0;
}

{

    int n,m,a,b;

}

//素数和
#include<stdio.h>
int main() {
    int n,i,sum,f,l,r;
    for(n=1; n<=100; n++) {
        sum=0;
        for(i=1; i<=2*n-1; i=i+2) {
            sum=sum+(n*n-n+i);
        }
        if(n*n*n==sum) {
            printf("%d*%d*%d=",n,n,n);
            f=sum/n;
            if(f%2==1) {
                l=f-(n-1)/2*2;
                r=f+(n-1)/2*2;
                while(l<=r) {
                    printf("%d+",l);
                    l=l+2;
                }
                printf("\b \n");
            }
            else {
                l=(f-1)-(n-2)/2*2;
                r=(f+1)+(n-2)/2*2;
                while(l<=r) {
                    printf("%d+",l);
                    l=l+2;
                }
                printf("\b \n");
            }
        }
    }
    return 0;
}

//菱形叠罗汉
#include<stdio.h>

int main() {
    int i,j,k,m,n;
    for (i=1; i<10; i++) {
        m = abs(5- i );
        n=i;
        if(i>5)
            n =10-i;
        j =2*n-1;
        for (k=1; k<=m; k++)
            printf(" ");
        for (k=1; k<=j; k++)
            printf("#");
        printf("\n");
    }
    return 0;
}
